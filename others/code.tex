% Euler method for the ODEs
clear all;  clf
h=0.1; x=0:h:1;
N=length(x)-1;
u(1)=0;
fun=@(t,u) t.^2+t-u;
for n=1:N
    u(n+1)=u(n)+h.*fun(x(n),u(n));
end
ue=-exp(-x)+x.^2-x+1;
plot(x,ue,'b-',x,u,'r+','LineWidth',1)
xlabel('x'), ylabel('u')
\end{lstlisting}

# Fibonacci series up to n
def fib(n):
    a, b = 0, 1
    while a < n:
        print(a, end=' ')
        a, b = b, a+b
    print()
fib(1000)
